/**
 * Fetch a paginated list of dinosaur genera from PBDB.
 
export async function fetchDinosaurs(offset = 0, limit = 50) {
  // Basic input normalization
  offset = Math.max(0, Number(offset) || 0);
  limit = Math.max(1, Math.min(200, Number(limit) || 50));

  try {
    // Use the public PBDB API. Adjust base_url if you need a different endpoint.
    const res = await fetch(
      "https://paleobiodb.org/data1.2/taxa/list.json?base_name=Dinosauria&rel=all_children&rank=genus&status=valid&limit=200",
    );

    if (!res.ok) {
      console.error("PBDB request failed with status", res.status);
      return [];
    }

    const data = await res.json();

    // PBDB structure: data.records is the array of results
    const records = Array.isArray(data?.records) ? data.records : [];

    // Remove synonym entries safely
    const filtered = records.filter((r) => {
      const name = r?.nam ?? "";
      return !name.toLowerCase().includes("synonym");
    });

    // Local pagination
    const page = filtered.slice(offset, offset + limit);

    // Normalize output
    return page.map((record, index) => ({
      id: record?.tid ?? offset + index,
      name: record?.nam ?? "Unknown",
      rank: record?.rnk ?? "genus",
      period: record?.tdf ?? "Mesozoic",
    }));
  } catch (err) {
    console.error("Fetch Dinosaurs failed:", err);
    return [];
  }
}
